    \begin{algorithm}[H]
        \DontPrintSemicolon
        
        \SetKwData{Stack}{\(S\)}
        \SetKwData{T}{t}
        \SetKwData{U}{u}

        \SetKwFunction{Closure}{closure}

        \ForEach{\( i = 1\) \KwTo \( | S | \)} {
            alreadyOn[i] = \textbf{false}
        }
        \Closure{\T, \Stack} {
            \Stack.push(\T)\;
            alreadyOn[\T] = \textbf{true}\;
            \ForEach{\U \(\in \textrm{move}_n(\textrm{\T}, \varepsilon)\)} {
                \If{\textbf{not} alreadyOn[\U]} {
                    \Closure{\U, \Stack}
                }
            }
        }
